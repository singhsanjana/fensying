\tikzset{
    %Define standard arrow tip
    >=stealth',
    %Define style for boxes
    punkt/.style={
           rectangle,
           minimum height=2em,
           text centered},
}

\begin{enumerate}
    \item \textit{X, Y are SC fences above and below instructions A, B respectively}\\
        \begin{tikzpicture}
            \node[punkt] (X) {X:\scriptsize $F_{sc}$};
            \node[punkt, below=of X] (A) {A:\scriptsize $W_x$}
                edge[-](X);
            \node[punkt, right=of X] (B) {B:\scriptsize $R_x$};
            \node[punkt, below=of B] (Y) {Y:\scriptsize $F_{sc}$}
            edge[-](B);
            
            \path[->,draw]
                (A) edge node[pos=0.3] {rf} (B);
     
        \end{tikzpicture}\\
        \rf{A}{B} \implies \tor{X}{Y}
    
    \item \textit{X, Y are SC fences below and above instructions A, B respectively. M is a previous write operation on variable x.}\\
        \begin{tikzpicture}
            \node[punkt] (A) {A:\scriptsize $W_x$};
            \node[punkt,below=of A] (X) {X:\scriptsize $F_{sc}$}
                edge[-] (A);
            \node[punkt,right=of A] (Y) {Y:\scriptsize $F_{sc}$};
            \node[punkt,below=of Y] (B) {B:\scriptsize $R_x$}
                edge[-] (Y);
            
        \end{tikzpicture}\\
        (\rf{M}{B}) \land\ (\mo{M}{A}) \implies \tor{Y}{X}
    
    \item \textit{X, Y are SC fences below and above instructions A, B respectively}\\
        \begin{tikzpicture}
            \node[punkt] (A) {A:\scriptsize $W_x$};
            \node[punkt,below=of A] (X) {X:\scriptsize $F_{sc}$}
                edge[-] (A);
            \node[punkt,right=of A] (Y) {Y:\scriptsize $F_{sc}$};
            \node[punkt,below=of Y] (B) {B:\scriptsize $W_x$}
                edge[-] (Y);

            \path[->,draw]
                (B) edge node[pos=0.3] {mo} (A);
        
        \end{tikzpicture}\\
        \mo{B}{A} \implies \tor{Y}{X}
        
\end{enumerate}
